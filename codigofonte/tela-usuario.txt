<script>
(function () {
    const L = 'pt';
    const base = '/' + L + '/api/support';
    const toggle = document.getElementById('support-toggle');
    const widget = document.getElementById('support-widget');
    const panel = document.getElementById('support-panel');
    const badge = document.getElementById('support-badge');
    const loadingEl = document.getElementById('support-loading');
    const formEl = document.getElementById('support-form');
    const chatEl = document.getElementById('support-chat');
    const threadEl = document.getElementById('support-thread');
    let ticketId = null, lastId = 0, pollTimer = null, isOpen = false;

    function esc(s) { const d = document.createElement('div'); d.textContent = s == null ? '' : s; return d.innerHTML; }

    function showOnly(el) {
        [loadingEl, formEl, chatEl].forEach(function (x) { x.classList.add('hidden'); });
        el.classList.remove('hidden');
    }

    function renderMsg(m) {
        const admin = m.isFromAdmin;
        const wrap = document.createElement('div');
        wrap.className = 'flex ' + (admin ? 'justify-start' : 'justify-end');
        wrap.innerHTML = '<div class="max-w-[80%] ' + (admin ? 'bg-slate-800 text-slate-200' : 'bg-primary/20 text-primary') + ' rounded-xl px-3 py-2">'
            + '<p class="text-sm leading-relaxed break-words">' + esc(m.content) + '</p>'
            + '<p class="text-[10px] mt-1 opacity-60">' + esc(m.createdAt) + '</p></div>';
        threadEl.appendChild(wrap);
        if (m.id && m.id > lastId) lastId = m.id;
    }

    function scrollBottom() { threadEl.scrollTop = threadEl.scrollHeight; }

    function fetchMessages() {
        if (!ticketId) return;
        fetch(base + '/ticket/' + ticketId + '/messages?after=' + lastId)
            .then(function (r) { return r.json(); })
            .then(function (d) {
                if (d.messages && d.messages.length) { d.messages.forEach(renderMsg); scrollBottom(); }
                if (d.status === 'closed') stopPoll();
            }).catch(function () {});
    }

    function startPoll() { stopPoll(); pollTimer = setInterval(fetchMessages, 5000); }
    function stopPoll() { if (pollTimer) { clearInterval(pollTimer); pollTimer = null; } }

    function openChat(id) {
        ticketId = id; lastId = 0; threadEl.innerHTML = '';
        showOnly(chatEl);
        fetchMessages();
        startPoll();
    }

    function loadState() {
        showOnly(loadingEl);
        fetch(base + '/active-ticket')
            .then(function (r) { return r.json(); })
            .then(function (d) { d.ticket ? openChat(d.ticket.id) : showOnly(formEl); })
            .catch(function () { showOnly(formEl); });
    }

    function refreshBadge() {
        fetch(base + '/active-ticket')
            .then(function (r) { return r.json(); })
            .then(function (d) {
                const n = d.ticket ? d.ticket.unread : 0;
                if (n > 0) { badge.textContent = n > 9 ? '9+' : n; badge.classList.remove('hidden'); }
                else badge.classList.add('hidden');
            }).catch(function () {});
    }

    function openPanel() { isOpen = true; panel.classList.remove('hidden'); widget.classList.add('hidden'); loadState(); }
    function closePanel() { isOpen = false; panel.classList.add('hidden'); widget.classList.remove('hidden'); stopPoll(); refreshBadge(); }

    toggle.addEventListener('click', openPanel);
    document.getElementById('support-close').addEventListener('click', closePanel);

    formEl.addEventListener('submit', function () {
        const subject = document.getElementById('support-subject').value.trim();
        const message = document.getElementById('support-message').value.trim();
        if (!subject || !message) return;
        fetch(base + '/ticket', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ subject: subject, message: message }) })
            .then(function (r) { return r.json(); })
            .then(function (d) {
                if (d.ticketId) {
                    document.getElementById('support-subject').value = '';
                    document.getElementById('support-message').value = '';
                    openChat(d.ticketId);
                }
            }).catch(function () {});
    });

    document.getElementById('support-reply-form').addEventListener('submit', function () {
        const input = document.getElementById('support-reply-input');
        const message = input.value.trim();
        if (!message || !ticketId) return;
        fetch(base + '/ticket/' + ticketId + '/message', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: message }) })
            .then(function (r) { return r.json(); })
            .then(function (d) { if (d.id) { renderMsg(d); scrollBottom(); input.value = ''; } })
            .catch(function () {});
    });

    refreshBadge();
    setInterval(function () { if (!isOpen) refreshBadge(); }, 30000);
})();
</script>

        <script>
    document.addEventListener('DOMContentLoaded', function() {
        // Add loading state to all form submit buttons
        document.querySelectorAll('form').forEach(function(form) {
            // Skip AJAX forms (those with onsubmit returning false)
            if (form.getAttribute('onsubmit')) return;
            // Skip forms in admin support chat
            if (form.id === 'admin-reply-form') return;
            // Skip support widget forms (handled by their own AJAX)
            if (form.id === 'support-form' || form.id === 'support-reply-form') return;
            // Skip AJAX chat forms (lobby match chat)
            if (form.id === 'chatForm' || form.id === 'mobileChatForm') return;

            form.addEventListener('submit', function(e) {
                const btn = form.querySelector('button[type="submit"], button:not([type])');
                if (!btn || btn.disabled) return;

                // Save original content
                btn.dataset.originalText = btn.innerHTML;
                btn.disabled = true;
                btn.classList.add('opacity-60', 'cursor-not-allowed');
                btn.innerHTML = '<span class="inline-flex items-center gap-2"><svg class="animate-spin h-4 w-4" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"></path></svg>' + (btn.dataset.originalText.length < 50 ? btn.dataset.originalText : 'Carregando...') + '</span>';

                // Re-enable after 10s in case of error
                setTimeout(function() {
                    btn.disabled = false;
                    btn.classList.remove('opacity-60', 'cursor-not-allowed');
                    if (btn.dataset.originalText) btn.innerHTML = btn.dataset.originalText;
                }, 10000);
            });
        });

        // Disable submit buttons when required fields are empty
        document.querySelectorAll('form').forEach(function(form) {
            const requiredInputs = form.querySelectorAll('input[required], select[required], textarea[required]');
            const btn = form.querySelector('button[type="submit"], button:not([type])');
            if (!btn || requiredInputs.length === 0) return;
            // Skip forms that already have their own validation logic
            if (form.id === 'forgotForm' || form.id === 'resetForm' || form.id === 'admin-reply-form') return;
            // Skip support widget forms (handled by their own AJAX)
            if (form.id === 'support-form' || form.id === 'support-reply-form') return;
            // Skip AJAX chat forms (lobby match chat)
            if (form.id === 'chatForm' || form.id === 'mobileChatForm') return;

            function checkRequired() {
                let allFilled = true;
                requiredInputs.forEach(function(input) {
                    if (input.type === 'checkbox') {
                        if (!input.checked) allFilled = false;
                    } else if (!input.value.trim()) {
                        allFilled = false;
                    }
                });
                btn.disabled = !allFilled;
                if (allFilled) {
                    btn.classList.remove('opacity-40', 'cursor-not-allowed');
                } else {
                    btn.classList.add('opacity-40', 'cursor-not-allowed');
                }
            }

            requiredInputs.forEach(function(input) {
                input.addEventListener('input', checkRequired);
                input.addEventListener('change', checkRequired);
            });

            checkRequired();
        });
    });
