<script>
(function() {
    var list = document.getElementById('defi-list');
    var listUrl = list.getAttribute('data-url');
    var listError = list.getAttribute('data-error');
    var currentFilter = 'all';
    var currentPage = 1;

    function loadChallenges(filter, page) {
        currentFilter = filter;
        currentPage = page || 1;
        var xhr = new XMLHttpRequest();
        xhr.open('GET', listUrl + '?filter=' + encodeURIComponent(filter) + '&page=' + currentPage + '&ajax=1');
        xhr.onload = function() {
            if (xhr.status === 200) {
                var tmp = document.createElement('div');
                tmp.innerHTML = xhr.responseText;
                var newList = tmp.querySelector('#defi-list');
                if (newList) {
                    list.innerHTML = newList.innerHTML;
                }
                list.scrollIntoView({ behavior: 'smooth', block: 'start' });
                bindPagination();
            } else {
                list.innerHTML = '<div class="text-center py-10 text-red-500">' + listError + '</div>';
            }
        };
        xhr.onerror = function() {
            list.innerHTML = '<div class="text-center py-10 text-red-500">' + listError + '</div>';
        };
        xhr.send();
    }

    // Tab filter clicks — use event delegation for iOS compatibility
    document.getElementById('defi-filters').addEventListener('click', function(e) {
        var btn = e.target.closest('button[data-filter]');
        if (!btn) return;
        e.preventDefault();
        var filter = btn.getAttribute('data-filter');
        // Update active tab styles
        var buttons = document.querySelectorAll('#defi-filters button[data-filter]');
        buttons.forEach(function(b) {
            b.classList.remove('bg-primary', 'text-white');
            b.classList.add('bg-slate-100', 'dark:bg-slate-800', 'text-slate-500', 'dark:text-slate-400');
        });
        btn.classList.add('bg-primary', 'text-white');
        btn.classList.remove('bg-slate-100', 'dark:bg-slate-800', 'text-slate-500', 'dark:text-slate-400');
        loadChallenges(filter, 1);
    });

    // Pagination clicks
    function bindPagination() {
        var pag = document.getElementById('defi-pagination');
        if (!pag) return;
        pag.addEventListener('click', function(e) {
            var btn = e.target.closest('.pagination-btn');
            if (!btn || btn.disabled) return;
            var page = parseInt(btn.getAttribute('data-page'));
            if (!page) return;
            loadChallenges(currentFilter, page);
        });
    }
    bindPagination();
})();

// Share challenge function
function shareChallenge(title, url) {
    // Try native Web Share API first (mobile)
    if (navigator.share) {
        navigator.share({ title: title, text: title + ' — The 1vs1', url: url }).catch(() => {});
        return;
    }
    // Fallback: show share popup
    const encoded = encodeURIComponent(url);
    const text = encodeURIComponent(title + ' — The 1vs1');
    const popup = document.getElementById('share-popup');
    popup.querySelector('[data-share="whatsapp"]').href = 'https://wa.me/?text=' + text + '%20' + encoded;
    popup.querySelector('[data-share="facebook"]').href = 'https://www.facebook.com/sharer/sharer.php?u=' + encoded;
    popup.querySelector('[data-share="twitter"]').href = 'https://twitter.com/intent/tweet?text=' + text + '&url=' + encoded;
    popup.querySelector('[data-share="telegram"]').href = 'https://t.me/share/url?url=' + encoded + '&text=' + text;
    popup.querySelector('[data-share="copy"]').onclick = function(e) {
        e.preventDefault();
        navigator.clipboard.writeText(url).then(() => {
            this.innerHTML = '<span class="material-symbols-outlined text-lg">check</span> Copiado!';
            setTimeout(() => { this.innerHTML = '<span class="material-symbols-outlined text-lg">content_copy</span> Copiar link'; }, 2000);
        });
    };
    popup.classList.remove('hidden');
}

<script>
(function() {
    var list = document.getElementById('defi-list');
    var listUrl = list.getAttribute('data-url');
    var listError = list.getAttribute('data-error');
    var currentFilter = 'all';
    var currentPage = 1;

    function loadChallenges(filter, page) {
        currentFilter = filter;
        currentPage = page || 1;
        var xhr = new XMLHttpRequest();
        xhr.open('GET', listUrl + '?filter=' + encodeURIComponent(filter) + '&page=' + currentPage + '&ajax=1');
        xhr.onload = function() {
            if (xhr.status === 200) {
                var tmp = document.createElement('div');
                tmp.innerHTML = xhr.responseText;
                var newList = tmp.querySelector('#defi-list');
                if (newList) {
                    list.innerHTML = newList.innerHTML;
                }
                list.scrollIntoView({ behavior: 'smooth', block: 'start' });
                bindPagination();
            } else {
                list.innerHTML = '<div class="text-center py-10 text-red-500">' + listError + '</div>';
            }
        };
        xhr.onerror = function() {
            list.innerHTML = '<div class="text-center py-10 text-red-500">' + listError + '</div>';
        };
        xhr.send();
    }

    // Tab filter clicks — use event delegation for iOS compatibility
    document.getElementById('defi-filters').addEventListener('click', function(e) {
        var btn = e.target.closest('button[data-filter]');
        if (!btn) return;
        e.preventDefault();
        var filter = btn.getAttribute('data-filter');
        // Update active tab styles
        var buttons = document.querySelectorAll('#defi-filters button[data-filter]');
        buttons.forEach(function(b) {
            b.classList.remove('bg-primary', 'text-white');
            b.classList.add('bg-slate-100', 'dark:bg-slate-800', 'text-slate-500', 'dark:text-slate-400');
        });
        btn.classList.add('bg-primary', 'text-white');
        btn.classList.remove('bg-slate-100', 'dark:bg-slate-800', 'text-slate-500', 'dark:text-slate-400');
        loadChallenges(filter, 1);
    });

    // Pagination clicks
    function bindPagination() {
        var pag = document.getElementById('defi-pagination');
        if (!pag) return;
        pag.addEventListener('click', function(e) {
            var btn = e.target.closest('.pagination-btn');
            if (!btn || btn.disabled) return;
            var page = parseInt(btn.getAttribute('data-page'));
            if (!page) return;
            loadChallenges(currentFilter, page);
        });
    }
    bindPagination();
})();

// Share challenge function
function shareChallenge(title, url) {
    // Try native Web Share API first (mobile)
    if (navigator.share) {
        navigator.share({ title: title, text: title + ' — The 1vs1', url: url }).catch(() => {});
        return;
    }
    // Fallback: show share popup
    const encoded = encodeURIComponent(url);
    const text = encodeURIComponent(title + ' — The 1vs1');
    const popup = document.getElementById('share-popup');
    popup.querySelector('[data-share="whatsapp"]').href = 'https://wa.me/?text=' + text + '%20' + encoded;
    popup.querySelector('[data-share="facebook"]').href = 'https://www.facebook.com/sharer/sharer.php?u=' + encoded;
    popup.querySelector('[data-share="twitter"]').href = 'https://twitter.com/intent/tweet?text=' + text + '&url=' + encoded;
    popup.querySelector('[data-share="telegram"]').href = 'https://t.me/share/url?url=' + encoded + '&text=' + text;
    popup.querySelector('[data-share="copy"]').onclick = function(e) {
        e.preventDefault();
        navigator.clipboard.writeText(url).then(() => {
            this.innerHTML = '<span class="material-symbols-outlined text-lg">check</span> Copiado!';
            setTimeout(() => { this.innerHTML = '<span class="material-symbols-outlined text-lg">content_copy</span> Copiar link'; }, 2000);
        });
    };
    popup.classList.remove('hidden');
}
</script>

        <script>
    document.addEventListener('DOMContentLoaded', function() {
        // Add loading state to all form submit buttons
        document.querySelectorAll('form').forEach(function(form) {
            // Skip AJAX forms (those with onsubmit returning false)
            if (form.getAttribute('onsubmit')) return;
            // Skip forms in admin support chat
            if (form.id === 'admin-reply-form') return;
            // Skip AJAX chat forms (lobby match chat)
            if (form.id === 'chatForm' || form.id === 'mobileChatForm') return;

            form.addEventListener('submit', function(e) {
                const btn = form.querySelector('button[type="submit"], button:not([type])');
                if (!btn || btn.disabled) return;

                // Save original content
                btn.dataset.originalText = btn.innerHTML;
                btn.disabled = true;
                btn.classList.add('opacity-60', 'cursor-not-allowed');
                btn.innerHTML = '<span class="inline-flex items-center gap-2"><svg class="animate-spin h-4 w-4" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"></path></svg>' + (btn.dataset.originalText.length < 50 ? btn.dataset.originalText : 'Carregando...') + '</span>';

                // Re-enable after 10s in case of error
                setTimeout(function() {
                    btn.disabled = false;
                    btn.classList.remove('opacity-60', 'cursor-not-allowed');
                    if (btn.dataset.originalText) btn.innerHTML = btn.dataset.originalText;
                }, 10000);
            });
        });

        // Disable submit buttons when required fields are empty
        document.querySelectorAll('form').forEach(function(form) {
            const requiredInputs = form.querySelectorAll('input[required], select[required], textarea[required]');
            const btn = form.querySelector('button[type="submit"], button:not([type])');
            if (!btn || requiredInputs.length === 0) return;
            // Skip forms that already have their own validation logic
            if (form.id === 'forgotForm' || form.id === 'resetForm' || form.id === 'admin-reply-form') return;
            // Skip AJAX chat forms (lobby match chat)
            if (form.id === 'chatForm' || form.id === 'mobileChatForm') return;

            function checkRequired() {
                let allFilled = true;
                requiredInputs.forEach(function(input) {
                    if (input.type === 'checkbox') {
                        if (!input.checked) allFilled = false;
                    } else if (!input.value.trim()) {
                        allFilled = false;
                    }
                });
                btn.disabled = !allFilled;
                if (allFilled) {
                    btn.classList.remove('opacity-40', 'cursor-not-allowed');
                } else {
                    btn.classList.add('opacity-40', 'cursor-not-allowed');
                }
            }

            requiredInputs.forEach(function(input) {
                input.addEventListener('input', checkRequired);
                input.addEventListener('change', checkRequired);
            });

            checkRequired();
        });
    });
    </script>

    if ('serviceWorker' in navigator) {

    // ── PWA Install Banner Logic ──
    (function() {
        const isStandalone = window.matchMedia('(display-mode: standalone)').matches || navigator.standalone;
        if (isStandalone) return;

        const isIOS = /iPhone|iPad|iPod/.test(navigator.userAgent) && !window.MSStream;
        const isAndroid = /Android/.test(navigator.userAgent);
        if (!isIOS && !isAndroid) return;

        const banner = document.getElementById('pwa-banner');
        const modal = document.getElementById('pwa-modal');
        const iosSection = document.getElementById('pwa-ios');
        const androidSection = document.getElementById('pwa-android');
        let deferredPrompt = null;

        // Show banner
        banner.classList.remove('hidden');
        banner.classList.add('flex');

        // Capture beforeinstallprompt for Android
        window.addEventListener('beforeinstallprompt', function(e) {
            e.preventDefault();
            deferredPrompt = e;
        });

        // Dismiss (session only — reappears on next page load)
        document.getElementById('pwa-dismiss-btn').addEventListener('click', function() {
            banner.classList.add('hidden');
        });

        // Open modal
        document.getElementById('pwa-install-btn').addEventListener('click', function() {
            modal.classList.remove('hidden');
            if (isIOS) {
                iosSection.classList.remove('hidden');
                androidSection.classList.add('hidden');
            } else {
                androidSection.classList.remove('hidden');
                iosSection.classList.add('hidden');
                if (!deferredPrompt) {
                    document.getElementById('pwa-native-install').classList.add('hidden');
                    document.getElementById('pwa-android-fallback').classList.remove('hidden');
                }
            }
        });

        // Close modal
        function closeModal() { modal.classList.add('hidden'); }
        document.getElementById('pwa-modal-close').addEventListener('click', closeModal);
        document.getElementById('pwa-modal-backdrop').addEventListener('click', closeModal);

        // Native Android install
        document.getElementById('pwa-native-install').addEventListener('click', function() {
            if (deferredPrompt) {
                deferredPrompt.prompt();
                deferredPrompt.userChoice.then(function() {
                    deferredPrompt = null;
                    closeModal();
                    banner.classList.add('hidden');
                });
            }
        });
    })();
    </script>
